Membuat List


# List Kosong
kosong = []

# List String
buah = ["apel", "jeruk", "mangga"]

# List Campuran
campur = [1, "dua", 3.0, True]

# List dari Range
angka = list(range(1, 6))  # [1, 2, 3, 4, 5]

Mengakses Elemen


buah = ["apel", "jeruk", "mangga", "pisang"]

# Index Positif (0, 1, 2...)
print(buah[0])   # apel
print(buah[2])   # mangga

# Index Negatif (-1, -2...)
print(buah[-1])  # pisang (Terakhir)

SLICING (Potongan)


# [start:stop]
print(buah[0:2])
# ["apel", "jeruk"]

print(buah[1:])
# ["jeruk", "mangga", "pisang"]

Modifikasi List


buah = ["apel", "jeruk", "mangga"]

# 1. Mengubah Data
buah[0] = "anggur"
# ["anggur", "jeruk", "mangga"]

# 2. Menambah Data (Append)
buah.append("pisang")
# ["anggur", "jeruk", "mangga", "pisang"]

# 3. Menyisipkan Data (Insert)
buah.insert(1, "nanas")
# ["anggur", "nanas", "jeruk", "mangga", "pisang"]

# 4. Menghapus Data (Remove)
buah.remove("jeruk")
# ["anggur", "nanas", "mangga", "pisang"]

Method List Populer

Method Fungsi
append(x) Tambah x di akhir
insert(i, x) Sisipkan x di index i
pop() Hapus & ambil elemen terakhir
remove(x) Hapus x pertama yang ditemukan
sort() Urutkan list (ascending)
reverse() Balik urutan list
count(x) Hitung jumlah kemunculan x

Fungsi Built-in

len(L) Panjang list
max(L) Nilai terbesar
min(L) Nilai terkecil
sum(L) Total jumlah 

angka = [5, 2, 8, 1, 9]

print(len(angka))  # 5
print(max(angka))  # 9
print(min(angka))  # 1
print(sum(angka))  # 25

Iterasi List (Looping)


buah = ["apel", "jeruk", "mangga"]

# Cara 1: Langsung (Recommended)
for item in buah:
    print(item)

# Cara 2: Pakai Index
for i in range(len(buah)):
    print(f"Index {i}: {buah[i]}")

# Cara 3: Enumerate (Index + Item)
for i, item in enumerate(buah):
    print(f"{i}: {item}")
